git --version - проверка версии

git connfig --global user.email "your@email.com"
git connfig --global user.name "your Name"

git connfig --list - проверка настроек

ssh-keygen - создали ssh ключ
cat ~/.ssh/id_rsa.pub - получение ssh ключа

cоздаем репозиторий на github
git init - превращает текущую директорию в git в репозиторий
git remote add название_связи (обычно - origin) ссылку_на_удаленный_репозиторый (ssh: git@github:...) 

git remote -v - просмотр списка удаленных репозиториев
giy remote rm название_связи (обычно - origin) - удаление связи 
git status - проверка статуса файлов
git add название_файлов - отслеживание конкретных файлов
git add . - отслеживание всех файлов


git commit - создание коммита (версии/слепка кода)
git commit -m 'описание версии' - создание коммита с комментарием

git push название_связи (обычно - origin) название_ветки - отправка версии кода на удаленный репозиторий
Например: git push origin master


git clone ссылку_на_удаленный_репозиторый - склонировать/скопировать/скачать удаленный репозиторий 

git diff - просмотр внесенных изменений

git branch - вывод списка веток в терминал (звездочкой помечается активная ветка)

git branch название_ветки - создание указанной ветки

git checkout название_ветки переход в указанную ветку
git swich название_ветки - переход в указанную ветку


git pull название_связи название_ветки - стягивание изменений с удаленного репозитория, с указанной ветки

git stash - спрятать локальные данные во временную область
git stash list - просмотр списка спрятанных данных
git stash apply - запись спрятанных данных/ извлечение спрятанных данных

git stash drop - удаление спрятанных данных

git log - просмотр коммитов/версий
git log --oneline - компактное отображение коммитов
